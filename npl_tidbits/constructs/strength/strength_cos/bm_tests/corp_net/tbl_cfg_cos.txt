#########################################################################################
#
# Purpose:
#   Configuration needed for 'strength' cos assignment example.
#
# Steps:
#  a. Populate pri_cos_mapping_table with following pattern.
#    >> pri_cos_mapping_table[i].cos = i,  cos_strength_index = i
#
#  b. Populate dscp_cos_mapping_table with following pattern
#    >> dscp_cos_mapping_table[0,..,7].cos = 0, cos_strength_index = 0
#    >> dscp_cos_mapping_table[8,..,15].cos = 1, cos_strength_index = 1
#      ......
#    >> dscp_cos_mapping_table[56,..,63].cos = 7, cos_strength_index = 7
#
#  c. Populate cos_strength_profile_table with follwoing pattern
#    >> cos_strength_profile_table[i].cos_strength = i
#   
#  With this configurations, winning object comparison is as following:
#    pri_cos_mapping_table[0].cos = dscp_cos_mapping_table[0-7].cos 
#      < local_bus.cos = pri_cos_mapping_table[1].cos = dscp_cos_mapping_table[8-15].cos 
#      < pri_cos_mapping_table[2].cos = dscp_cos_mapping_table[16-23].cos
#      < pri_cos_mapping_table[3].cos = dscp_cos_mapping_table[24-31].cos
#      < pri_cos_mapping_table[4].cos = dscp_cos_mapping_table[32-39].cos
#      < pri_cos_mapping_table[5].cos = dscp_cos_mapping_table[40-47].cos
#      < pri_cos_mapping_table[6].cos = dscp_cos_mapping_table[48-55].cos
#      < pri_cos_mapping_table[7].cos = dscp_cos_mapping_table[56-63].cos
#
#########################################################################################

# pri_cos_mapping_table configuration
for i=0,7 'lt pri_cos_mapping_table insert _INDEX=$i cos=$i cos_strength_index=$i'

# dscp_cos_mapping_table configuration
for i=0,7 'lt dscp_cos_mapping_table insert _INDEX=$i cos=0 cos_strength_index=0'
for i=8,15 'lt dscp_cos_mapping_table insert _INDEX=$i cos=1 cos_strength_index=1'
for i=16,23 'lt dscp_cos_mapping_table insert _INDEX=$i cos=2 cos_strength_index=2'
for i=24,31 'lt dscp_cos_mapping_table insert _INDEX=$i cos=3 cos_strength_index=3'
for i=32,39 'lt dscp_cos_mapping_table insert _INDEX=$i cos=4 cos_strength_index=4'
for i=40,47 'lt dscp_cos_mapping_table insert _INDEX=$i cos=5 cos_strength_index=5'
for i=48,55 'lt dscp_cos_mapping_table insert _INDEX=$i cos=6 cos_strength_index=6'
for i=56,63 'lt dscp_cos_mapping_table insert _INDEX=$i cos=7 cos_strength_index=7'

# cos_strength_profile_table configuration
for i=0,7 'lt cos_strength_profile_table insert _INDEX=$i cos_strength=$i'
